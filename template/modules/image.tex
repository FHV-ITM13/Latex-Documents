% Module image
\usepackage{eso-pic}
\usepackage{graphicx}

% coderef command
\newcommand{\imgref}[1]{\ref{\imglabelprefix:#1}}

% images
\newcommand{\img}[4]{
    \begin{figure}[!ht]
  		\begin{center}
    		\includegraphics[width=#1\textwidth]{#2}
    		\caption{#3}
    		\label{\imglabelprefix:#4}
    	\end{center}
    \end{figure}
}
%% e.g.:
% \img{} %% filename in figures folder
%       {width=0.5\textwidth,height=0.5\textheight}%% maximum width/height, aspect ratio will be kept
%       {}%% caption
%       {}%% label
\newcommand{\imgsrc}[5]{
    \begin{figure}[!ht]
  		\begin{center}
    		\includegraphics[width=#1\textwidth]{#2}
    		\caption{#3 \\ Quelle: #5}
    		\label{\imglabelprefix:#4}
    	\end{center}
    \end{figure}
}
%% e.g.:
% \img{} %% filename in figures folder
%       {width=0.5\textwidth,height=0.5\textheight}%% maximum width/height, aspect ratio will be kept
%       {}%% caption
%       {}%% label
%       {}%% source - displayed below caption text

\newcommand{\imgcite}[5]{
    \begin{figure}[!ht]
  		\begin{center}
    		\includegraphics[width=#1\textwidth]{#2}
    		\caption{#3 \cite{#5}}
    		\label{\imglabelprefix:#4}
    	\end{center}
    \end{figure}
}
%% e.g.:
% \imgcite{} %% filename in figures folder
%       {width=0.5\textwidth,height=0.5\textheight}%% maximum width/height, aspect ratio will be kept
%       {}%% caption
%       {}%% label
%       {}%% source - will be shown in references
